14def _is_in_preparation(event): 

15 return not event.is_public and now() <= event.datetime_from 

16 

17 

18def _is_cfp_open(event): 

19 return not _is_in_preparation(event) and event.cfp.is_open

22def _is_in_review(event): 

23 return ( 

24 not _is_cfp_open(event) 

25 and event.submissions.filter(state=SubmissionStates.SUBMITTED).exists() 

26 and now() <= event.datetime_from 

27 )

30def _is_in_scheduling_stage(event): 

31 return ( 

32 not _is_running(event) and not _is_in_wrapup(event) and not _is_in_review(event) 

33 )

36def _is_running(event): 

37 return event.datetime_from <= now() <= event.datetime_to

40def _is_in_wrapup(event): 

41 return event.datetime_to <= now()

123def in_stage(event, stage): 

124 return STAGES[stage]["method"](event)

127def build_event_url(event, url): 

128 result = event 

129 for part in url: 

130 result = getattr(result, part) 

131 return result

134def get_stages(event): 

135 inactive_state = "done" 

136 stages = copy.deepcopy(STAGES) 

137 

138 for stage in STAGES: 

139 is_stage_active = inactive_state == "done" and in_stage(event, stage) 

140 if is_stage_active: 

141 inactive_state = "todo" 

142 stages[stage]["phase"] = "current" if is_stage_active else inactive_state 

143 for link in stages[stage].get("links", []): 

144 if "url" in link and link["url"]: 

145 link["url"] = build_event_url(event, link["url"]) 

146 return stages
