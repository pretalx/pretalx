12def path_with_hash(name, base_path=None, max_length=100): 

13 base_path = base_path or "" 

14 dir_name, file_name = os.path.split(name) 

15 file_root, file_ext = os.path.splitext(file_name) 

16 file_root = safe_filename(file_root) 

17 random = get_random_string(7) 

18 if base_path and max_length: 

19 # We need to resolve the base path for its actual total length, as absolute 

20 # paths are stored in the database. 

21 full_base_path = Path(settings.MEDIA_ROOT) / base_path 

22 total_length = len( 

23 str(full_base_path / dir_name / f"{file_root}_{random}{file_ext}") 

24 ) 

25 if total_length > max_length: 

26 # If the total length of the path exceeds the max length, we need to 

27 # shorten the file name by the difference. 

28 file_root = file_root[: -(total_length - max_length)] 

29 return str(Path(base_path) / dir_name / f"{file_root}_{random}{file_ext}")
