13def test_widget_data_cached_response_preserves_content_type( 

14 client, public_event_with_schedule 

15): 

16 """Cached widget responses must keep Content-Type: application/json. 

17 

18 On the first (cold-cache) request, widget_data() returns a JsonResponse 

19 with the correct Content-Type. On the second request the response is 

20 served from cache via etag_cache_page(), which must preserve the 

21 Content-Type rather than falling back to the default text/html. 

22 

23 Regression: etag_cache_page wraps the cached JsonResponse in a plain 

24 HttpResponse, discarding all headers including Content-Type. With 

25 text/html, HTML minification middleware can garble the JSON body. 

26 """ 

27 event = public_event_with_schedule 

28 url = event.urls.schedule_widget_data 

29 

30 cold_response = client.get(url) 

31 assert cold_response.status_code == 200 

32 assert cold_response["Content-Type"] == "application/json" 

33 cold_data = cold_response.json() 

34 

35 hot_response = client.get(url) 

36 assert hot_response.status_code == 200 

37 assert hot_response["Content-Type"] == "application/json" 

38 hot_data = hot_response.json() 

39 

40 assert hot_data == cold_data
