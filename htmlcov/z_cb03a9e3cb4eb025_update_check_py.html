28def run_update_check(sender, **kwargs): 

29 gs = GlobalSettings()

31 return

40@app.task(name="pretalx.common.update_check") 

41@scopes_disabled() 

42def update_check(): 

43 gs = GlobalSettings() 

44 

45 if not gs.settings.update_check_enabled: 

46 return 

47 

48 if not gs.settings.update_check_id: 

49 gs.settings.set("update_check_id", uuid.uuid4().hex) 

50 

51 if "runserver" in sys.argv or "devserver" in sys.argv: # pragma: no cover 

52 gs.settings.set("update_check_last", now()) 

53 gs.settings.set("update_check_result", {"error": "development"}) 

54 return 

55 

56 check_payload = { 

57 "id": gs.settings.update_check_id, 

58 "version": pretalx_version, 

59 "events": { 

60 "total": Event.objects.count(), 

61 "public": Event.objects.filter(is_public=True).count(), 

62 }, 

63 "plugins": [ 

64 {"name": plugin.module, "version": plugin.version} 

65 for plugin in get_all_plugins() 

66 ], 

67 } 

68 import requests 

69 

70 try: 

71 response = requests.post( 

72 "https://pretalx.com/.update_check/", 

73 json=check_payload, 

74 timeout=30, 

75 ) 

76 except requests.RequestException: # pragma: no cover 

77 gs.settings.set("update_check_last", now()) 

78 gs.settings.set("update_check_result", {"error": "unavailable"}) 

79 return 

80 

81 gs.settings.set("update_check_last", now()) 

82 if response.status_code != 200: 

83 gs.settings.set("update_check_result", {"error": "http_error"}) 

84 return 

85 

86 rdata = response.json() 

87 update_available = rdata["version"]["updatable"] or any( 

88 plugin["updatable"] for plugin in rdata["plugins"].values() 

89 ) 

90 gs.settings.set("update_check_result_warning", update_available) 

91 if update_available and rdata != gs.settings.update_check_result: 

92 send_update_notification_email() 

93 gs.settings.set("update_check_result", rdata)

127def check_result_table(): 

128 gs = GlobalSettings() 

129 res = gs.settings.update_check_result 

130 if not res: 

131 return {"error": "no_result"} 

132 

133 if "error" in res: 

134 return res

136 table = [ 

137 ( 

138 "pretalx", 

139 pretalx_version, 

140 res["version"]["latest"], 

141 res["version"]["updatable"], 

142 ) 

143 ] 

144 for plugin in get_all_plugins(): 

145 if plugin.module in res["plugins"]: 

146 pdata = res["plugins"][plugin.module] 

147 table.append( 

148 ( 

149 _("Plugin") + f": {plugin.name}", 

150 plugin.version, 

151 pdata["latest"], 

152 pdata["updatable"], 

153 ) 

154 ) 

155 else: 

156 table.append((_("Plugin") + f": {plugin.name}", plugin.version, "?", False))
