16@scopes_disabled() 

17def shortlink_view(request, code, *args, **kwargs): 

18 with suppress(Submission.DoesNotExist): 

19 submission = Submission.objects.select_related("event").get(code=code) 

20 if request.user.has_perm("submission.orga_list_submission", submission): 

21 return HttpResponseRedirect(submission.orga_urls.base) 

22 if request.user.has_perm("submission.view_public_submission", submission): 

23 return HttpResponseRedirect(submission.urls.public) 

24 with suppress(User.DoesNotExist): 

25 user = User.objects.get(code=code) 

26 if request.user.has_perm("person.administrator_user", user): 

27 return HttpResponseRedirect(user.orga_urls.admin) 

28 profiles = user.profiles.select_related("event").order_by("-created") 

29 for profile in profiles: 

30 if request.user.has_perm("person.orga_list_speakerprofile", profile): 

31 return HttpResponseRedirect(profile.orga_urls.base) 

32 if profile := profiles.first(): 

33 if request.user == user: 

34 return HttpResponseRedirect(profile.event.urls.user) 

35 if request.user.has_perm("person.view_speakerprofile", profile): 

36 return HttpResponseRedirect(profile.urls.public) 

37 raise Http404()
