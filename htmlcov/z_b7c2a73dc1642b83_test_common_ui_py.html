4import pytest

6from pretalx.common.ui import has_good_contrast

9@pytest.mark.parametrize( 

10 "color,expected", 

11 ( 

12 # Dark colors - should have good contrast with white text 

13 ("#000000", True), # Black - 21:1 contrast 

14 ("#1a1a1a", True), # Very dark grey 

15 ("#0000ff", True), # Blue - 8.59:1 contrast 

16 ("#800000", True), # Maroon 

17 ("#800080", True), # Purple 

18 # Medium colors - may or may not pass depending on exact shade 

19 ("#3aa57c", False), # pretalx green - ~2.99:1, doesn't pass 4.5 

20 ("#008000", True), # Dark green - 5.14:1, passes 

21 # Light colors - should NOT have good contrast with white text 

22 ("#ffffff", False), # White 

23 ("#ffff00", False), # Yellow 

24 ("#00ffff", False), # Cyan 

25 ("#f0f0f0", False), # Light grey 

26 ("#ffcc00", False), # Gold/amber 

27 ("#90ee90", False), # Light green 

28 ("#add8e6", False), # Light blue 

29 # Edge cases around the 4.5 threshold 

30 ("#767676", True), # Just passes WCAG AA (4.54:1) 

31 ("#777777", False), # Just fails WCAG AA (4.48:1) 

32 ), 

33) 

34def test_has_good_contrast(color, expected): 

35 assert has_good_contrast(color) == expected
