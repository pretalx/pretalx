29class PretalxSerializer(ModelSerializer): 

30 """ 

31 This serializer class will behave like the i18nfield serializer, 

32 outputting a dict/object for internationalized strings, unless if 

33 when it was initialized with an ``override_locale`` (taken from 

34 a URL queryparam, usually), in which case the string will be cast 

35 to the locale in question – relying on either a view or a middleware 

36 to apply the locale manager. 

37 """ 

38 

39 def __init__(self, *args, **kwargs): 

40 self.override_locale = kwargs.get("context", {}).get("override_locale") 

41 self.event = getattr(kwargs.get("context", {}).get("request"), "event", None) 

42 super().__init__(*args, **kwargs) 

43 

44 def get_with_fallback(self, data, key): 

45 """ 

46 Get key from dictionary, or fall back to `self.instance` if it exists. 

47 Handy for validating data in partial updates. 

48 (Yes, not terribly safe, but better than nothing.) 

49 """ 

50 if key in data: 

51 return data[key] 

52 if self.instance: 

53 return getattr(self.instance, key, None) 

54 

55 @cached_property 

56 def extra_flex_field_config(self): 

57 return { 

58 key: split_levels(self._flex_options_all[key]) 

59 for key in ("expand", "fields", "omit") 

60 } 

61 

62 def get_extra_flex_field(self, extra_field, *args, **kwargs): 

63 if extra_field in self.extra_flex_field_config["expand"][0]: 

64 klass, settings = self.Meta.extra_expandable_fields[extra_field] 

65 serializer_class = self._get_serializer_class_from_lazy_string(klass) 

66 settings["context"] = self.context 

67 settings["parent"] = self 

68 for key, value in self.extra_flex_field_config.items(): 

69 if value[1] and extra_field in value[1]: 

70 settings[key] = value[1][extra_field] 

71 return serializer_class(*args, **settings, **kwargs)
