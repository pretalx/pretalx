4"""Django tries to be helpful by suggesting to run "makemigrations" in 

5alarmingly red font on every "migrate" run. 

6 

7Since pretalx migrations are most often run by pretalx users, that is – 

8administrators, this is dangerous advice. Running "makemigrations" on a 

9project that you intend to continue to install from PyPI will lead to no 

10end of headaches, so by this very dirty hack, we're removing the 

11warnigns from user-facing output. 

12""" 

13 

14import sys 

15 

16from django.core.management.base import OutputWrapper 

17from django.core.management.commands.migrate import Command as Parent 

18 

19 

20class OutputFilter(OutputWrapper): 

21 banlist = ( 

22 "have changes that are not yet reflected", 

23 "Run 'manage.py makemigrations' to make new ", 

24 ) 

25 

26 def write(self, msg, style_func=None, ending=None): 

27 if any(word in msg for word in self.banlist): 

28 return 

29 super().write(msg, style_func, ending) 

30 

31 

32class Command(Parent): 

33 def __init__(self, stdout=None, stderr=None, no_color=False, force_color=False): 

34 super().__init__(stdout, stderr, no_color, force_color) 

35 self.stdout = OutputFilter(stdout or sys.stdout)
