78 def __str__(self) -> str: 

79 """Used in choice drop downs.""" 

80 if not self.default_duration: 

81 return str(self.name) 

82 if self.default_duration > 60 * 24: 

83 return _("{name} ({duration} days)").format( 

84 name=self.name, 

85 duration=pleasing_number(round(self.default_duration / 60 / 24, 1)), 

86 ) 

87 if self.default_duration > 90: 

88 return _("{name} ({duration} hours)").format( 

89 name=self.name, 

90 duration=pleasing_number(round(self.default_duration / 60, 1)), 

91 ) 

92 return _("{name} ({duration} minutes)").format( 

93 name=self.name, duration=self.default_duration 

94 )

113 :class:`~pretalx.submission.models.submission.Submission` objects of 

114 this type. 

115
